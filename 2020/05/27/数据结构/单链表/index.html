以链式结构存储的线性表称为链表
存储单元可以是相邻的，也可以是不相邻的
链表的每一个节点只包含一个指针与，因此又称为单链

结构

数据域 指针域
指针域存放着下一个数据的头地址

非空链表与空链表

e.g.

存储地址 数据域 指针域
1 李 40
7 张 40
15 赵 Null
20 王 1
40 钱 14
14 孙 13
13 吴 7

操作集

定义链表

struct node
{
	int data;
	struct node* next;
}

初始化

struct node* initlink()
{
	struct node* p;
	p = (struct node*)malloc(sizeof(struct node));
	p->next = null;
	return p;
}

malloc申请指定空间的内存
sizeof返回一个变量或内存的大小

填入

void insert(struct node* l)
{
	int x;
	struct node* p;
	while(1)
	{
		printf("请输入一个数据：");
		scannf_s("%d", &x);
		if(x == 0)
		{
			break;
		}
		p = (struct node*)malloc(sizeof(struct node));
		p->data = x;
		p->next = l->next;
		l->next = p;
	}
}

填入的数据呈反向输出

输出

void show(struct node* l)
{
	struct node* p;
	p = l;
	while(p->next != null)
	{
		p = p->next;
		printf("%-4d", p->data);
	}
}

返回长度

int length(struct node* l)
{
	int len = 0;
	struct node* p;
	p = l;
	while(p->next != 0)
	{
		p = p->next;
		length++;
	}
	return len;
}

临时变量p是必须的，否则链表数据会丢失

查找

void search(struct node* l)
{
	int x;
	struct node* p;
	int i = 0;
	p = l;
	printf("请输入要查找的数：");
	scannf_s("%d", &x);
	while(p->next != null)
	{
		p = p->next;
		i++;
		if(p->data == x)
		{
			break;
		}
	}
	if(p != null)
	{
		printf("位置在：%d\n", i);
	}
	else
	{
		printf("未找到");
	}
}
